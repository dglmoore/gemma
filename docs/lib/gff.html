} else if (end < 1) { this.error(`end value is less than 1, got "${end}"`, line); } return end; }, parseScore(field, line) { const score = (field === '.') ? field : parseInt(field); if (score !== '.' && isNaN(score)) { this.error(`score is not a number, got "${field}"`, line);
